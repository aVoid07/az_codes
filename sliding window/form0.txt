#include <bits/stdc++.h>
using namespace std;
int freq[1000100];
int dist =0;
void solution(){
    int n, k;
    cin>>n>>k;
    int arr[n];
    for(int i = 0; i < n;i++) cin >> arr[i];
    int ans = 1e9;
    for(int i = 0; i < k; i++){
        if(freq[arr[i]] == 0) dist++;
        freq[arr[i]]++;
    }
    int tail = 0, head = k-1;
    while(head < n){
        while(head + 1 < n && (freq[arr[head+1]] > 0 || dist < k)){
            head++;
            if(freq[arr[head]] == 0) dist++;
            freq[arr[head]]++;
        }
        ans = min(ans, head - tail + 1);
        head++;
        tail++;
    }
    return ans;
}
int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
      solution();
    }
    return 0;
}
